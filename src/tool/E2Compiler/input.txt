@name Button
@outputs Out
@persist Plys:array
@model models/cheeze/buttons/button_reset.mdl

if(first()){
    function void playSound(Path:string){
        entity():soundPlay(randint(9999),1,Path)
    }
    
    function number entity:check(){
        foreach(K,E:entity = Plys){
            if(E == This){
                playSound("buttons/button14.wav")
                return 1
            }
        }
        
        playSound("buttons/blip1.wav")
        return 0
    }
    
    runOnUse(1)
    
    Plys = array(owner())
}

if(useClk()){
#ifdef moneyTimeoutTitle()
    local Temp = useClk()
#else
    local Temp = useEntClk():inUseBy()
#endif
    
    if(Temp:check()){
        Out = 1
        
        timer("stop",10)
    }
}

if(clk("stop")){
    Out = 0
}
